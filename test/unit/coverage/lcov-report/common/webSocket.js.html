/**
 * 将发送websocket封装
 * */
let send = function (ws, struct) {
  let sw = new streamWriter();
  let data = struct.getData();
  sw.createBuffer(struct.getBufferLen());
  sw.writeBuffer(data.data, data.format);
  ws.send(sw.getBuffer());
}
 
/**
 * 解析二进制数据为字符串
 * */
let convertByteToStruct = function (byteData, bagName) {
  if (bagName == null) bagName = "Header";
  let obj = new WSData.DataBag[bagName].struct();
  let sr = new streamReader(byteData, obj.getFormat());
  obj.setData(sr.readToEnd());
  return obj;
}
/****
 * 解析数据包头
 * */
let getResponseData = function (struct, head, reader) {
  let name = struct.toString();
  let retVal = null;
 
  if ((head.getCommandH() & WSData.Command.CMD_RECORDSETS) == WSData.Command.CMD_RECORDSETS) {
    retVal = [];
    let ds = new WSData.DataBag.DSHeader.struct();
    reader.setFormat(ds.getFormat());
    ds.setData(reader.readToEnd());
 
    reader.setFormat(struct.getFormat());
    for (let i = 0, len = ds.getRowCntS(); i < len; ++i) {
      retVal.push(getResponseData_Fill(reader, name));
    }
  } else {
    reader.setFormat(struct.getFormat());
    retVal = getResponseData_Fill(reader, name);
    if (head.getSubcmdH() == WSData.Subcmd.UP_WARN_MSG_ADPT_INFO) {
      retVal.plateId = head.getSourceH();
    }
  }
  return retVal;
}

let getResponseData_Fill = function (reader, structName) {
  let wsObj = (new WSData.Original[structName]()).getWSObject();
  let row = reader.readToEnd();
  let j = 0;
  for (let item in wsObj) {
    if (row[j].toString().indexOf('%') >= 0) wsObj[item] = decodeFromGb2312(row[j]);
    else wsObj[item] = row[j];
    j++;
  }
  return wsObj;
}

let getalarmvehiclelist = function (ret, reader) {
  let vehicle = {
    model: '',
    type: '',
  }, type = ret.subcmdH.value;
  if (type == 324) {
    type = "overspeed";
    let struct = new WSData.Struct.WarnInfoOverSpeed();
    let data = {
      type: 1,
      data: getResponseData(struct, ret, reader)
    }
    vehicle.type = type;
    vehicle.model = data.data;
    vehicle_list.push(vehicle);
    return vehicle_list;
  }
  if (type == 325) {
    type = "abnormalmove";
    let struct = new WSData.Struct.WarnInfoNightAbnormal();
    let data = {
      type: 1,
      data: getResponseData(struct, ret, reader)
    }
    vehicle.type = type;
    vehicle.model = data.data;
    vehicle_list.push(vehicle);
    return vehicle_list;
  }
  if (type == 326) {
    type = "over4hour";
    let struct = new WSData.Struct.WarnInfoFourHours();
    let data = {
      type: 1,
      data: getResponseData(struct, ret, reader)
    }
    vehicle.type = type;
    vehicle.model = data.data;
    vehicle_list.push(vehicle);
    return vehicle_list;
  }
}
